Subgrid pro zarovnání formulářových polí:
Volitelné 
/* Rodičovský grid definuje sloupce I řádky */
.products-grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  grid-template-rows: auto;   /* řádky se tvoří automaticky */
  gap: var(--base-3);
}

/* Každá karta "obalí" 5 řádků rodiče */
.product-card {
  grid-row: span 5;           /* rezervuj 5 řádků v rodiči */
  display: grid;
  grid-template-rows: subgrid; /* použij ty stejné řádky */
  /* → obrázek, název, popis, cena, tlačítko jsou zarovnány
        napříč kartami bez ohledu na délku obsahu */
}

/* Bez subgrid: každá karta má vlastní vnitřní grid
   → obsah se nezarovná přes karty */
.product-card--no-subgrid {
  display: flex;
  flex-direction: column;
}
