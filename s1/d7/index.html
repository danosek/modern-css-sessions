Container query — základ

/* 1. Označ kontejner */
.card-wrapper {
  container-type: inline-size;  /* sleduje inline rozměr */
  container-name: product;      /* volitelné jméno */
}

/* 2. Piš query vůči kontejneru, ne viewportu */
@container product (min-width: 400px) {
  .product-card {
    display: grid;
    grid-template-columns: 140px 1fr;  /* přejdi na horizontální layout */
  }
}

@container product (min-width: 560px) {
  .product-card__footer {
    display: flex;
    justify-content: space-between;
    align-items: center;
  }
}

/* Kdy použít co */
@media (min-width: 768px)        { /* viewport — layout stránky, viditelnost */ }
@container (min-width: 400px)    { /* kontejner — vnitřní komponenta */         }

Range Container Queries (cross-browser, Chrome 111+, FF 110+, Safari 16+)

/* Místo min-width / max-width — čitelnější range syntax */
@container product (width < 360px) {
  /* stack layout — výchozí (základní styl) */
}

@container product (360px <= width < 560px) {
  .product-card {
    display: grid;
    grid-template-columns: 140px 1fr;
  }
}

@container product (width >= 560px) {
  .product-card {
    display: grid;
    grid-template-columns: 160px 1fr;
  }
  .product-card__footer {
    flex-direction: row;
    justify-content: space-between;
    align-items: center;
  }
}

/* Ekvivalent v tradiční syntaxi — pro srovnání */
/* @container product (min-width: 360px)                   { ... } */
/* @container product (min-width: 360px) and
              (max-width: 559px)                           { ... } */
