/* 1. Označ kontejner */
.card-wrapper {
  container-type: inline-size;  /* sleduje inline rozměr */
  container-name: product;      /* volitelné jméno */
}

/* 2. Piš query vůči kontejneru, ne viewportu */
@container product (min-width: 400px) {
  .product-card {
    display: grid;
    grid-template-columns: 140px 1fr;  /* přejdi na horizontální layout */
  }
}

@container product (min-width: 560px) {
  .product-card__footer {
    display: flex;
    justify-content: space-between;
    align-items: center;
  }
}

/* Kdy použít co */
@media (min-width: 768px)        { /* viewport — layout stránky, viditelnost */ }
@container (min-width: 400px)    { /* kontejner — vnitřní komponenta */         }

/* ─── Style Container Queries (experimentální, Chrome 111+) ─── */

/* 1. Označ kontejner — sleduje custom property */
.theme-card {
  container-type: inline-size;
  container-name: theme-card;
  --variant: default;           /* výchozí hodnota */
}

/* 2. JS mění custom property na elementu */
card.style.setProperty('--variant', 'featured');

/* 3. @container style() reaguje na hodnotu */
@container style(--variant: featured) {
  .theme-card__body   { background: var(--surface-brand-primary-subtle); }
  .theme-card__inner  { box-shadow: 0 0 0 2px var(--surface-brand-primary-strong); }
}

@container style(--variant: muted) {
  .theme-card__inner  { opacity: 0.6; }
}

/* Výhoda oproti JS třídám: logika stylu zůstává v CSS, JS jen nastaví data */
