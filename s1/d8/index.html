/* Rodičovský grid definuje sloupce I řádky */
.products-grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  grid-template-rows: auto;   /* řádky se tvoří automaticky */
  gap: var(--base-3);
}

/* Každá karta "obalí" 5 řádků rodiče */
.product-card {
  grid-row: span 5;           /* rezervuj 5 řádků v rodiči */
  display: grid;
  grid-template-rows: subgrid; /* použij ty stejné řádky */
  /* → obrázek, název, popis, cena, tlačítko jsou zarovnány
        napříč kartami bez ohledu na délku obsahu */
}

/* Bez subgrid: každá karta má vlastní vnitřní grid
   → obsah se nezarovná přes karty */
.product-card--no-subgrid {
  display: flex;
  flex-direction: column;
}

/* ─── Editoriální layout — 4 zóny zarovnané přes 3 karty ─── */

/* Rodičovský grid: 3 sloupce + 4 explicitní řádky */
.editorial-grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  grid-template-rows: repeat(4, auto);  /* obrázek, nadpis, perex, CTA */
  gap: var(--base-3);
}

/* Každá karta "vypůjčí" 4 řádky a předá je dětem přes subgrid */
.editorial-card {
  grid-row: span 4;
  display: grid;
  grid-template-rows: subgrid;   /* → obrázek/nadpis/perex/CTA aligned */
}

/* Výsledek: CTA tlačítka všech karet sedí na stejné výšce
   i když mají nadpisy a perexy různé délky — bez JS, bez fixed výšek */
