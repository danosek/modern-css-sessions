Container query — základ

/* 1. Označ kontejner */
.card-wrapper {
  container-type: inline-size;  /* sleduje inline rozměr */
  container-name: product;      /* volitelné jméno */
}

/* 2. Piš query vůči kontejneru, ne viewportu */
@container product (min-width: 400px) {
  .product-card {
    display: grid;
    grid-template-columns: 140px 1fr;  /* přejdi na horizontální layout */
  }
}

@container product (min-width: 560px) {
  .product-card__footer {
    display: flex;
    justify-content: space-between;
    align-items: center;
  }
}

/* Kdy použít co */
@media (min-width: 768px)        { /* viewport — layout stránky, viditelnost */ }
@container (min-width: 400px)    { /* kontejner — vnitřní komponenta */         }
